DROP DATABASE leadplan;
COMMIT;
CREATE DATABASE leadplan CHARACTER SET utf8mb4;
COMMIT;
USE leadplan;
create table SYAKENINFO
( 
1CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
PLATENO VARCHAR (20) NOT NULL ,
FIRSTDATE VARCHAR (30) NOT NULL ,
BODYNO VARCHAR (30) NOT NULL ,
TYPEDIV VARCHAR (10) NOT NULL ,
KINDDIV VARCHAR (10) NOT NULL ,
EXPIRYDATE DATE NOT NULL ,
4CODE BIGINT NOT NULL ,
3CODE BIGINT NOT NULL 
);

create table ZAIKOINFO
( 
2CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
BUYPRICE VARCHAR (10) NOT NULL ,
BUYTAX VARCHAR (9) NOT NULL ,
CARRECYCLE VARCHAR (9) NOT NULL ,
BUYCOST VARCHAR (9) NOT NULL ,
CARTAX VARCHAR (9) NOT NULL ,
4CODE BIGINT NOT NULL ,
BUYDATE DATE NOT NULL ,
MAKEDATE VARCHAR (20) NOT NULL 
);

create table USERINFO
( 
3CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (3CODE),
USERNAME VARCHAR (50) NOT NULL ,
USERADD1 VARCHAR (200) NOT NULL ,
USERPOSTCD VARCHAR (7) NOT NULL ,
USERTELNO VARCHAR (15) NOT NULL ,
USERFAXNO VARCHAR (15) NOT NULL ,
USERMAIL VARCHAR (150) NOT NULL 
);

create table CARINFO
( 
4CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
CARNAME VARCHAR (150) NOT NULL ,
CARGRADE VARCHAR (150) NOT NULL 
);

create table SYAKENFILEINFO
( 
5CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (5CODE),
SFILEPATH VARCHAR (255) NOT NULL ,
1CODE BIGINT NOT NULL 
);

create table KIROKUBOINFO
( 
6CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (6CODE),
KFILEPATH VARCHAR (255) NOT NULL ,
1CODE BIGINT NOT NULL 
);

create table CARENDERINFO
( 
7CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
LUSERID BIGINT (10) NOT NULL ,
SUBJECT VARCHAR (255) NOT NULL ,
MAINTEXT VARCHAR (255) NOT NULL ,
SCHEDATE DATE NOT NULL ,
ADDATE DATE NOT NULL 
);

create table LOGINUSERINFO 
(
LUSERID BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
LUSERNAME VARCHAR (50) NOT NULL ,
LUSERPASS VARCHAR (50) NOT NULL );
create table SYSTEMINFO 
(
ORNERID BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
STARTDATE DATE NOT NULL );
COMMIT;
