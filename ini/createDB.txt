DROP DATABASE kani;
COMMIT;
CREATE DATABASE kani CHARACTER SET cp932;
COMMIT;
USE kani;
create table sizaiinfo
( 
1CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (1CODE),
SIZAIID VARCHAR (4) NOT NULL UNIQUE ,
SIZAINAME VARCHAR (60) NOT NULL ,
SAIKB VARCHAR (1) NOT NULL 
);

create table kokyakuinfo
( 
2CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (2CODE),
KOKYAKUID VARCHAR (3) NOT NULL ,
KOKYAKUNAME VARCHAR (60) NOT NULL 
);

create table zaikoinfo
( 
3CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (3CODE),
1CODE BIGINT NOT NULL ,
ZAIKONUM VARCHAR (7) NOT NULL DEFAULT '0' 
);

create table genbainfo
( 
4CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (4CODE),
GENBAID VARCHAR (5) NOT NULL ,
GENBANAME VARCHAR (60) NOT NULL ,
2CODE BIGINT NOT NULL ,
GENBASTATUS VARCHAR (1) NOT NULL DEFAULT '0' ,
ENDDATE DATE 
);

create table syukkainfo
( 
5CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (5CODE),
4CODE BIGINT NOT NULL ,
1CODE BIGINT (4) NOT NULL ,
SYUKKANUM VARCHAR (7) NOT NULL 
);

create table henkyakuinfo
( 
6CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (6CODE),
4CODE BIGINT (5) NOT NULL ,
1CODE BIGINT (4) NOT NULL ,
HENKYAKUNUM VARCHAR (7) NOT NULL 
);

create table rirekiinfo
( 
7CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (7CODE),
CREATEDATE DATE NOT NULL ,
SAGYOUDATE DATE NOT NULL ,
4CODE BIGINT NOT NULL ,
2CODE BIGINT NOT NULL ,
1CODE BIGINT NOT NULL ,
IONUM VARCHAR (7) NOT NULL ,
IOTYPE VARCHAR (1) NOT NULL 
);

create table saiinfo
( 
8CODE BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
PRIMARY KEY (8CODE),
4CODE BIGINT NOT NULL ,
2CODE BIGINT NOT NULL ,
1CODE BIGINT NOT NULL ,
SAITYPE VARCHAR (1) NOT NULL ,
REASON VARCHAR (60) ,
SAICREATEDATE DATETIME NOT NULL ,
SAIUPDATE DATETIME ,
SAISTATUS VARCHAR (1) NOT NULL ,
SAINUM BIGINT NOT NULL 
);

create table loginuserinfo 
(
LUSERID BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
LUSERNAME VARCHAR (50) NOT NULL ,
LUSERPASS VARCHAR (50) NOT NULL );
create table systeminfo 
(
ORNERID BIGINT NOT NULL UNIQUE AUTO_INCREMENT ,
STARTDATE DATE NOT NULL );
COMMIT;
